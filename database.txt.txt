-- DROP SCHEMA public;

CREATE SCHEMA public AUTHORIZATION pg_database_owner;
-- public.alunos definição

-- Drop table

-- DROP TABLE public.alunos;

CREATE TABLE public.alunos (
	id int4 NOT NULL,
	nome varchar(70) NOT NULL,
	ativo bool DEFAULT true NULL,
	CONSTRAINT alunos_pk PRIMARY KEY (id)
);


-- public.disciplinas definição

-- Drop table

-- DROP TABLE public.disciplinas;

CREATE TABLE public.disciplinas (
	id int4 NOT NULL,
	nome varchar(40) NOT NULL,
	ativo bool DEFAULT true NOT NULL,
	CONSTRAINT disciplinas_pk PRIMARY KEY (id)
);


-- public.matriculas definição

-- Drop table

-- DROP TABLE public.matriculas;

CREATE TABLE public.matriculas (
	id int4 NOT NULL,
	fk_id_turma int4 NOT NULL,
	fk_id_estudante int4 NOT NULL,
	CONSTRAINT matriculas_pk PRIMARY KEY (id)
);


-- public.professores definição

-- Drop table

-- DROP TABLE public.professores;

CREATE TABLE public.professores (
	id int4 NOT NULL,
	nome varchar(70) NOT NULL,
	cpf varchar(11) NOT NULL,
	ativo bool DEFAULT true NOT NULL,
	CONSTRAINT professores_pk PRIMARY KEY (id),
	CONSTRAINT professores_unique UNIQUE (cpf)
);


-- public.turmas definição

-- Drop table

-- DROP TABLE public.turmas;

CREATE TABLE public.turmas (
	id int4 NOT NULL,
	fk_id_professor int4 NOT NULL,
	fk_id_disciplina int4 NOT NULL,
	ativo bool DEFAULT true NOT NULL,
	CONSTRAINT turmas_pk PRIMARY KEY (id)
);
